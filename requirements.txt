import streamlit as st
import pandas as pd
import matplotlib.pyplot as plt

# Title and description
st.title("Your Web App Title")
st.write("This app visualizes data interactively.")

# Load dataset directly
data_file = "Assignment 2_PastYearSales.csv"  # Specify the path to your data file
data = pd.read_csv(data_file, encoding='latin-1')
print(data.head())

# Display the dataset
st.write("Dataset Preview:")
st.dataframe(data.head())

# Example visualization
st.write("Visualization Example:")
fig, ax = plt.subplots()
data.plot(kind="bar", x='month', y='Sales (# of units)', ax=ax)
st.pyplot(fig)
